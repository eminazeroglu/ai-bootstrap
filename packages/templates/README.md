```
│   ├── mistakes-log.md.tpl
│   ├── verified-facts.md.tpl

│       ├── az.md.tpl
│       ├── ru.md.tpl
│       ├── en.md.tpl
│       └── tr.md.tpl

│   ├── ai-studio/
│   ├── brand-site/
│   ├── social-ops/
│   └── data-platform/

```
